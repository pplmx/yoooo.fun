def create_data_set():
    group = array([[1.0, 1.1], [1.0, 1.0], [0, 1], [0, 0.1]])

    return group, labels


def classify(in_x, data_set, labels, k):
    """
    åˆ†ç±»å™¨
    :param in_x: ç”¨äºåˆ†ç±»çš„è¾“å…¥å‘é‡
    :param data_set: è¾“å…¥çš„è®­ç»ƒæ ·æœ¬é›†
    :param labels: æ ‡ç­¾å‘é‡
    :param k: ç”¨äºé€‰æ‹©æœ€è¿‘é‚»å±…çš„æ•°ç›®
    :return:
    """
    # è·å–data_setçš„ç¬¬ä¸€ç»´é•¿åº¦
    data_set_size = data_set.shape[0]
    # åˆ†åˆ«è®¡ç®—è¾“å…¥å‘é‡ä¸data_seté›†åˆä¸­å„ç‚¹çš„å‘é‡å·®,å¹¶å­˜å…¥æ•°ç»„ä¸­
    diff_arr = tile(in_x, (data_set_size, 1)) - data_set
    # å¹³æ–¹
    sq_diff_arr = diff_arr ** 2
    # æ±‚å¹³æ–¹å’Œ
    sq_distinces = sq_diff_arr.sum(axis=1)
    # å¼€æ ¹,å¾—å„ç‚¹ä¸è¾“å…¥å‘é‡çš„è·ç¦»å€¼é›†åˆ
    distinces = sq_distinces ** 0.5
    # æ’åº,å‡åº(è¿”å›ç»“æœä¸ºç´¢å¼•,å¦‚[17,23,1,0],æ’åºåè¿”å›[3,2,0,1])
    sorted_dist_indices = distinces.argsort()
    # print('æœ€è¿‘çš„ç‚¹:%s' % labels[sorted_dist_indices[0]])
    # å­˜å‚¨æœ€è¿‘çš„kä¸ªç‚¹
    class_count = {}
    for i in range(k):
        vote_label = labels[sorted_dist_indices[i]]
        class_count[vote_label] = class_count.get(vote_label, 0) + 1
    # print(class_count)
    # æ ¹æ®å­—å…¸class_countçš„valueè¿›è¡Œé™åºæ’åˆ—
    # åœ¨æœ€è¿‘ç‚¹æ¡ˆä¾‹ä¸­,valueéƒ½æ˜¯1,ä¸‹é¢çš„æ’åºç­‰äºæ²¡åš
    sorted_class_count = sorted(class_count.items(),
                                key=operator.itemgetter(1), reverse=True)
    # print(sorted_class_count)
    return sorted_class_count[0][0]


def file2array(filename):
    # è·å–æ–‡ä»¶è¡Œæ•°
    fr = open(filename)
    array_lines = fr.readlines()
    amount = len(array_lines)
    # åˆ›å»ºç¬¬ä¸€ç»´é•¿åº¦ä¸ºamount,ç¬¬äºŒç»´é•¿åº¦ä¸º3çš„é›¶å¡«å……æ•°ç»„
    # å› ä¸ºæˆ‘ä»¬éœ€è¦å­˜å‚¨çš„æ•°æ®ç¬¬äºŒç»´é•¿åº¦ä¸º3,æ•…æˆ‘ä»¬è®¾ç½®å›ºå®šé•¿åº¦3
    return_array = zeros((amount, 3))
    # åˆ›å»ºç©ºlist
    class_label_vector = []
    index = 0
    for line in array_lines:
        # strip([chars]) å»é™¤å¤´å°¾çš„å­—ç¬¦,é»˜è®¤å»é™¤ç©ºæ ¼å­—ç¬¦
        line = line.strip()
        list_from_line = line.split('\t')
        # è·å–å‰ä¸‰ä¸ªå…ƒç´ ,èµ‹å€¼ç»™return_array(è¿™é‡Œé‡‡ç”¨äº†å¤šç»´åˆ‡ç‰‡)
        return_array[index, :] = list_from_line[0:3]
        # è´Ÿç´¢å¼•ä»åè·å–,-1è·å–æœ€åä¸€ä¸ªå…ƒç´ 
        class_label_vector.append(int(list_from_line[-1]))
        index += 1
    # è¿”å›tuple
    return return_array, class_label_vector

def auto_norm(data_set):
    """
    å½’ä¸€åŒ–ç‰¹å¾å€¼:è‡ªåŠ¨å°†æ•°æ®é›†è½¬åŒ–ä¸º0åˆ°1åŒºé—´å†…çš„å€¼
    ç”±äºé‡Œç¨‹æ•°è¿œè¿œå¤§äºå…¶ä»–ç‰¹å¾å€¼,å¯¹ç»“æœå½±å“è¿‡å¤§
    è€ŒHelenè®¤ä¸ºä¸‰è€…åŒç­‰é‡è¦,æ•…é‡‡ç”¨å½’ä¸€åŒ–å¤„ç†
    :param data_set:
    :return:
    """
    # å–ç¬¬ä¸€ç»´åº¦çš„æœ€å°å€¼
    """
        >>> sh = array([
                [[1, 1],[8, 18],[100, 3],[2, 4]],
                [[1, 90],[21, 2],[11, 3],[19, 4]]
            ])
        >>> shape(sh)
        (2,4,2)
        >>> sh.max()
        100
        >>> sh.min()
        1
        >>> sh.max(0)
        array([[  1,  90],
               [ 21,  18],
               [100,   3],
               [ 19,   4]])
        >>> sh.min(0)
        array([[ 1,  1],
               [ 8,  2],
               [11,  3],
               [ 2,  4]])
    """
    min_vals = data_set.min(0)
    max_vals = data_set.max(0)
    ranges = max_vals - min_vals
    # åˆ›å»ºä¸€ä¸ªä¸data_setå„ç»´é•¿åº¦å‡ç›¸åŒçš„é›¶å¡«å……æ•°ç»„
    # norm_data_set = zeros(shape(data_set))
    length = data_set.shape[0]
    norm_data_set = data_set - tile(min_vals, (length, 1))
    norm_data_set = norm_data_set / tile(ranges, (length, 1))
    return norm_data_set, ranges, min_vals


def dating_class_test():
    """
    æµ‹è¯•ä»£ç 
    :return:
    """
    # ç”¨äºæµ‹è¯•çš„æ•°æ®,å æ€»æ•°æ®çš„ç™¾åˆ†æ¯”
    # [å¦‚:å·²æœ‰100æ¡æ•°æ®,å…¶ä¸­90æ¡ä½œä¸ºæ ·æœ¬è®­ç»ƒæ•°æ®,å‰©ä½™10æ¡ç”¨äºæµ‹è¯•ç®—æ³•,æ£€æµ‹ç®—æ³•çš„æ­£ç¡®ç‡]
    test_ratio = 0.10
    dating_data_arr, dating_labels = file2array('resource/datingTestSet2.txt')
    norm_arr, ranges, min_vals = auto_norm(dating_data_arr)
    length = norm_arr.shape[0]
    # æµ‹è¯•æ•°æ®æ€»æ•°
    num_test_data = int(length * test_ratio)
    error_count = 0.0
    for i in range(num_test_data):
        classifier_result = classify(norm_arr[i, :], norm_arr[num_test_data:length, :],
                                     dating_labels[num_test_data:length], 3)
        print('The classifier came back with: %d, the real answer is: %d'
              % (classifier_result, dating_labels[i]))
        # å¦‚æœåˆ†ç±»å™¨ç»“æœå’ŒçœŸå®æ•°æ®,ä¸åŒ;error_count+1
        if classifier_result != dating_labels[i]:
            error_count += 1.0
    print('The total error rate is: %f' % (error_count / num_test_data))


def classify_person():
    """
    é¢„æµ‹å‡½æ•°
    :return:
    """
    result_list = ['not like', 'general like', 'very like']
    game_percent = float(input('percentage of time spent playing video games?'))
    fly_miles = float(input('frequent flier miles earned per year?'))
    how_much_ice_cream = float(input('liters of ice cream consumed per week?'))
    dating_data_arr, dating_labels = file2array('resource/datingTestSet2.txt')
    norm_arr, ranges, min_vals = auto_norm(dating_data_arr)
    in_arr = array([fly_miles, game_percent, how_much_ice_cream])
    classifier_result = classify((in_arr - min_vals) / ranges, norm_arr, dating_labels, 3)
    print('You will probably like this person: %s' % result_list[classifier_result - 1])


def img2vector(filename):
    """
    å°†32*32çš„äºŒè¿›åˆ¶å›¾åƒçŸ©é˜µè½¬åŒ–ä¸º1*1024çš„å‘é‡
    :param filename:
    :return:
    """
    return_vector = zeros((1, 1024))
    fr = open(filename)
    for i in range(32):
        line = fr.readline()
        for j in range(32):
            return_vector[0, 32*i+j] = int(line[j])
    return return_vector
