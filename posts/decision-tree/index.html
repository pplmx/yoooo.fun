def calc_shannon_entropy(data_set):
    num_entries = len(data_set)
    label_counts = {}
    # å¯¹å„ç±»åˆ«å‡ºç°çš„æ¬¡æ•°,è¿›è¡Œç»Ÿè®¡
    for feat_vector in data_set:
        current_label = feat_vector[-1]
        if current_label not in label_counts.keys():
            label_counts[current_label] = 0
        label_counts[current_label] += 1
    shannon_entropy = 0.0
    for key in label_counts:
        # è®¡ç®—è¯¥åˆ†ç±»çš„æ¦‚ç‡
        probability = label_counts[key] / num_entries
        # é€šè¿‡å¾ªç¯,å°†å„åˆ†ç±»çš„ä¿¡æ¯æœŸæœ›å€¼ç›¸åŠ 
        shannon_entropy -= probability * log(probability, 2)
    # è¿”å›é¦™å†œç†µ
    return shannon_entropy


def calc_shannon_entropy2(data_set):
    """
        é€šè¿‡åˆ—è¡¨æ¨å¯¼å¼,åŠCounter,å®ç°é¦™å†œç†µçš„è®¡ç®—
    :param data_set:
    :return:
    """
    # å–å‡º'yes','yes','no'ç­‰æ•°æ®æ”¾åˆ°æ•°ç»„ä¸­
    class_count = [sample[-1] for sample in data_set]
    # è·å–æ•°æ®é›†é•¿åº¦
    length = len(data_set)
    # å¯¹'yes','no'ç­‰å„ç±»åˆ«å‡ºç°çš„æ¬¡æ•°,è¿›è¡Œç»Ÿè®¡
    class_count = Counter(class_count)
    shannon_entropy = 0.
    # è®¡ç®—é¦™å†œç†µ
    for times in class_count.values():
        shannon_entropy -= times / length * log(times / length, 2)
    return shannon_entropy


def split_data_set(data_set, axis, value):
    """
        åˆ’åˆ†æ•°æ®é›†
    :param data_set: å¾…åˆ’åˆ†çš„æ•°æ®é›†
    :param axis: åˆ’åˆ†æ•°æ®é›†çš„ç‰¹å¾
    :param value: éœ€è¦è¿”å›çš„ç‰¹å¾çš„å€¼
    :return:
    """
    """
        appendå’Œextendçš„åŒºåˆ«
        >>> a = [1,2,3]
        >>> b = [4,5,6]
        >>> a.append(b)
        >>> a
        [1, 2, 3, [4, 5, 6]]
        >>> a.extend(b)
        >>> a
        [1, 2, 3, [4, 5, 6], 4, 5, 6]
    """
    divided_data_set = []
    for feature_vector in data_set:
        # if true,å°±å°†è¯¥å€¼remove,åŒæ—¶æ·»åŠ è¿›divided_data_set
        if feature_vector[axis] == value:
            reduced_feature_vector = feature_vector[:axis]
            reduced_feature_vector.extend(feature_vector[axis + 1:])
            divided_data_set.append(reduced_feature_vector)
    return divided_data_set


def choose_best_feature2split(data_set):
    # è·å–ç‰¹å¾å€¼çš„æ•°é‡
    num_features = len(data_set[0]) - 1
    # è®¡ç®—åŸå§‹é¦™å†œç†µ
    base_entropy = calc_shannon_entropy2(data_set)
    # æœ€ä½³ä¿¡æ¯å¢ç›Š
    best_info_gain = 0.
    # æœ€ä½³ç‰¹å¾å€¼çš„ä½ç½®ç´¢å¼•
    best_feature = -1
    for i in range(num_features):
        # åˆ›å»ºå”¯ä¸€çš„åˆ†ç±»æ ‡ç­¾åˆ—è¡¨
        feature_list = [example[i] for example in data_set]
        unique_vals = set(feature_list)
        # è®¡ç®—æ¯ç§åˆ’åˆ†æ–¹å¼çš„ä¿¡æ¯ç†µ
        new_entropy = 0.
        for value in unique_vals:
            sub_data_set = split_data_set(data_set, i, value)
            probability = len(sub_data_set) / len(data_set)
            new_entropy += probability * calc_shannon_entropy2(sub_data_set)
        # è®¡ç®—æœ€å¥½çš„ä¿¡æ¯å¢ç›Š
        # print('åŸå§‹ä¿¡æ¯ç†µä¸º%f' % base_entropy)
        # print('æ–°çš„ä¿¡æ¯ç†µä¸º%f' % new_entropy)
        info_gain = base_entropy - new_entropy
        # print('æŒ‰ç…§ç¬¬%dä¸ªç‰¹å¾å±æ€§åˆ’åˆ†,ä¿¡æ¯å¢ç›Šä¸º%f' % (i, info_gain))
        if info_gain > best_info_gain:
            best_info_gain = info_gain
            best_feature = i
    # print('æ•…æœ€ä½³ç‰¹å¾å±æ€§çš„ç´¢å¼•ä¸º%d' % best_feature)
    return best_feature


def majority_counter(class_list):
    """
        å¦‚æœæ•°æ®é›†å·²ç»å¤„ç†äº†æ‰€æœ‰å±æ€§,ä½†æ˜¯ç±»æ ‡ç­¾ä¾ç„¶ä¸æ˜¯å”¯ä¸€çš„,æˆ‘ä»¬éœ€è¦å†³å®šå¦‚ä½•å®šä¹‰è¯¥å¶å­èŠ‚ç‚¹
        æ­¤æ—¶,æˆ‘ä»¬é‡‡ç”¨å¤šæ•°è¡¨å†³çš„æ–¹å¼,å†³å®šè¯¥å¶å­èŠ‚ç‚¹çš„åˆ†ç±»
    :param class_list:
    :return:
    """
    class_count = {}
    for vote in class_list:
        if vote not in class_count.keys():
            class_count[vote] = 0
        class_count[vote] += 1
    sorted_class_count = sorted(class_count.items(), key=operator.itemgetter(1), reverse=True)
    return sorted_class_count[0][0]


def classify(input_tree, feature_labels, test_vector):
    first_str = list(input_tree.keys())[0]
    second_dict = input_tree[first_str]
    # å°†æ ‡ç­¾å­—ç¬¦ä¸²è½¬æ¢ä¸ºç´¢å¼•
    feature_index = feature_labels.index(first_str)
    class_label = None
    for key in second_dict.keys():
        if test_vector[feature_index] == key:
            if type(second_dict[key]).__name__ == 'dict':
                class_label = classify(second_dict[key], feature_labels, test_vector)
            else:
                class_label = second_dict[key]
    return class_label

def create_tree(data_set, labels):
    # è·å–ç±»åˆ«åˆ—è¡¨
    class_list = [example[-1] for example in data_set]
    # ç±»åˆ«å®Œå…¨ç›¸åŒ,åˆ™åœæ­¢ç»§ç»­åˆ’åˆ†
    if class_list.count(class_list[0]) == len(class_list):

        return class_list[0]

    if len(data_set[0]) == 1:

    # æœ€ä½³ç‰¹å¾å±æ€§çš„ç´¢å¼•
    best_feature = choose_best_feature2split(data_set)
    # æœ€ä½³ç‰¹å¾æ ‡è®°
    best_feature_label = labels[best_feature]
    # åˆ›å»ºå­—å…¸,å­˜å‚¨å†³ç­–æ ‘

        my_tree[best_feature_label][value] = create_tree(split_data_set(data_set, best_feature, value), sub_labels)
